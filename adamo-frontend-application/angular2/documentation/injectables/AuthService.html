  public login(email: string, password: string) {
    this.logger.debug(`login @${this.BACKEND_URI}/auth/login with ${email} and ${password}`);
    return (
      this.http.post(
          `${this.BACKEND_URI}/auth/login`,
          {
            email: email,
            password: password
          },
          {
            withCredentials: true
          }
        )
        .pipe(
          map( (response: LoginPayloadDto) => {
            // login successful if there's a jwt token in the response
            this.logger.debug(`user = ${response}`);
            this.logger.debug(`user.token = ${response.token}`);
            if (response.user && response.token) {
              // store user details and jwt token in local storage to keep user logged in between page refreshes
              localStorage.setItem("adamo.currentUser", JSON.stringify(response.user));
              this.currentUserSubject.next(response.user);
            }
            return response.user;
          })
        )
    );
  }

  public logout() {
    // remove user from local storage to log user out
    localStorage.removeItem("adamo.currentUser");
    this.currentUserSubject.next(null);
    return this.http.get(this.BACKEND_URI + "/logout", {
        withCredentials: true
      })
  }
